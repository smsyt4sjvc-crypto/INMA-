      <p>The <strong>Producer Price Index (PPI)</strong> is published monthly by the U.S. Bureau of Labor Statistics. It tracks what manufacturers charge for goods before they reach consumers. function calculateScore() {
  var curStr = document.getElementById('ppiCurrent').value.trim();
  var avgStr = document.getElementById('ppiAverage').value.trim();

  var current = parseFloat(curStr);
  if (isNaN(current) || current <= 0) {
    alert('Please enter a valid PPI value.');
    return;
  }

  var avg = parseFloat(avgStr);
  if (isNaN(avg) || avg <= 0) {
    avg = current * 0.98;
  }

  var pctDiff = ((current - avg) / avg) * 100;
  var score = 5 + (pctDiff / 2);
  score = Math.max(1, Math.min(10, Math.round(score * 10) / 10));

  var box = document.getElementById('resultBox');
  var scoreEl = document.getElementById('resultScore');
  var labelEl = document.getElementById('resultLabel');
  var descEl = document.getElementById('resultDesc');
  var marker = document.getElementById('resultMarker');

  scoreEl.textContent = score.toFixed(1);

  box.className = 'result-box show';

  if (score <= 3.5) {
    box.classList.add('result-favorable');
    labelEl.textContent = 'Favorable &#8212; Below Average';
    descEl.textContent = 'Material costs are trending below their recent averages. Good conditions for starting a project. This pricing window tends to close when demand picks up.';
  } else if (score <= 6.5) {
    box.classList.add('result-normal');
    labelEl.textContent = 'Normal &#8212; Holding Steady';
    descEl.textContent = 'Material costs are holding near their recent average. Stable conditions for planning and pricing. No rush, no reason to wait.';
  } else {
    box.classList.add('result-expensive');
    labelEl.textContent = 'Expensive &#8212; Above Average';
    descEl.textContent = 'Materials are running above their recent norms. Projects started now will likely cost more. Consider locking in quotes sooner rather than later.';
  }

  var markerPct = ((score - 1) / 9) * 100;
  marker.style.left = markerPct + '%';

  box.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
}

document.getElementById('ppiCurrent').addEventListener('keydown', function(e) {
  if (e.key === 'Enter') calculateScore();
});
document.getElementById('ppiAverage').addEventListener('keydown', function(e) {
  if (e.key === 'Enter') calculateScore();
});
document.querySelector('.mobile-toggle').addEventListener('click', function() {
  document.getElementById('navLinks').classList.toggle('open');
});
